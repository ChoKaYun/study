$(function(){
	var examBox3 = $('.exam_wrap:eq(2)'),
		orderList3 = examBox3.find('ul li'),
		thisBtnPlus = examBox3.find('button:eq(0)'),
		thisBtnMinus = examBox3.find('button:eq(1)'),
		currentIndex3 = 1;
	var changColor = function() {
		orderList3.eq(currentIndex3).css('background', 'skyblue');
		orderList3.eq(currentIndex3 - 1).removeAttr('style');
		currentIndex3 += 1;
	}
	var changColor2 = function() {
		currentIndex3 -= 1;
		orderList3.eq(currentIndex3).removeAttr('style');
		orderList3.eq(currentIndex3 - 1).css('background', 'skyblue');
	}
	thisBtnPlus.on('click', changColor);
	thisBtnMinus.on('click', changColor2);
}

$(function(){
	var examBox4 = $('.exam_wrap:eq(3)'),
		orderList4 = examBox4.find('ul li'),
		addClass = examBox4.find('button:eq(0)'),
		removeClass = examBox4.find('button:eq(1)'),
		currentIndex4 = 1;
	var changClass = function() {
		orderList4.eq(currentIndex4).addClass('active');
		orderList4.eq(currentIndex4 - 1).removeClass('active');
		currentIndex4 += 1;
	}
	var changClass2 = function() {
		currentIndex4 -= 1;
		orderList4.eq(currentIndex4).removeClass('active');
		orderList4.eq(currentIndex4 - 1).addClass('active');
	}
	addClass.on('click', changClass);
	removeClass.on('click', changClass2);
}
