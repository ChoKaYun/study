Q. 처음으로 일치하는 엘리먼트를 반환하시오.

1em
2em
3em

Tip) :first / .first()

$(function(){
	$('.exam_wrap:eq(0)').find('button').on('click', function(){
		$('.exam_wrap:eq(0)').find('.exam_q div').first().css('background-color', 'skyblue');
	});
}
